program Test;

function fib(x:integer):integer;
begin
  if (x = 0) then
    fib:=1;
  if (x = 1) then
    fib:=1
  else
    fib:=fib(x - 2) + fib(x - 1);
end;

function odd(n : integer): integer;
begin
  if (n = 0) then
      odd := 0
  else
      odd := even(n-1);
end;

function even(n : integer): integer;
begin
  if (n = 0) then
      even:=1
  else
      even := odd(n-1);
end;
